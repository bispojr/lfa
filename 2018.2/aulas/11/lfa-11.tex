	\begin{frame}{Questão 037}	
		\begin{block}{[Q037]}
			Sobre o um AFN $M$, é \underline{incorreto} afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item para $M$ aceitar $\omega$, é necessário que todos os ramos de execução aceitem $\omega$.
			\item a sua função $\delta$ tenha como saída um conjunto de estados.
			\item a sua função de $\delta$ tem como uma de suas entradas um símbolo de $\Sigma_{\epsilon}$.
			\item $M$ tem apenas um estado inicial.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 041}	
		\begin{block}{[Q041]}
			Na definição formal de computação para um AFN $N$, se $N$ aceita $\omega$, então existe uma sequência de estados $r_0, r_1, \ldots, r_m$ em que
			\begin{itemize}
				\item - $r_0 = q_0$;
				\item - $\delta(r_i, \omega_{i+1}) \in r_{i+1}$, para $i=0, \ldots, m-1$, e
				\item - $r_m \in F$.
			\end{itemize}
			O que o valor de $m$ representa?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item a quantidade de estados da sequência
			\item o tamanho da cadeia $\omega$
			\item a quantidade de entradas distintas de $\delta$
			\item a quantidade de estados de $N$
		\end{enumerate}
	\end{frame}
