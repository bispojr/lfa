	\begin{frame}{Questão 042}	
		\begin{block}{[Q042]}
			É verdade que todo AFN tem um AFD equivalente. Na prova apresentada pelo Sipser, ele constroi um AFD $M$ a partir de um AFN $N$. Se $N$ tem 10 estados, quantos estados teria $M$?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item 10
			\item 100
			\item $2^{10}$
			\item $10^2$
		\end{enumerate}
	\end{frame}
	
	\begin{frame}{Questão 043}	
		\begin{block}{[Q043]}
			É verdade que todo AFN tem um AFD equivalente. Na prova apresentada pelo Sipser, ele constroi um AFD $M = (Q', \Sigma, \delta', q_0', F')$ a partir de um AFN $N = (Q, \Sigma, \delta, q_0, F)$. 
			
			\vspace*{0.3cm}
						
			$Q' = \mathcal{P}(Q)$ porque...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item sempre um AFD tem mais estados que um AFN.
			\item $\mathcal{P}(Q)$ é o contradomínio de $\delta$.
			\item $\mathcal{P}(Q)$ é o conjunto de estados de $N$.
			\item o conjunto vazio é subconjunto de qualquer conjunto.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 044}	
		\begin{block}{[Q044]}
			É verdade que todo AFN tem um AFD equivalente. Na prova apresentada pelo Sipser, ele constroi um AFD $M = (Q', \Sigma, \delta', q_0', F')$ a partir de um AFN $N = (Q, \Sigma, \delta, q_0, F)$. 
			
			\vspace*{0.3cm}
						
			$F' = \{R \in Q'$ | $R$ contém um estado de aceitação de $N\}$ porque...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item é possível que $Q' = Q$, então é necessário explicitar os estados finais.
			\item se $R \in F'$ então todos os estados que estão em $R$ são finais.
			\item $R$ representa o nível da árvore de execução de N em que pelo menos um dos estados é final.
			\item $R$ necessita ser um estado de $Q$, e não um conjunto de estados de $Q$.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 045}	
		\begin{block}{[Q045]}
			É verdade que todo AFN tem um AFD equivalente. Na prova apresentada pelo Sipser, ele constroi um AFD $M = (Q', \Sigma, \delta', q_0', F')$ a partir de um AFN $N = (Q, \Sigma, \delta, q_0, F)$. 
			
			\vspace*{0.3cm}
			
			A função $\delta'(R,a) = \bigcup\limits_{r \in R} \delta(r,a)$ não está devidamente representada porque...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item não inclui os estados alcançados por possíveis transições $\epsilon$.
			\item não considera todos os símbolos distintos de $\Sigma$.
			\item utiliza $\bigcup\limits_{r \in R} \delta(r,a)$ ao invés de $\bigcap\limits_{r \in R} \delta(r,a)$.
			\item porque $R$ não é um estado de $Q$.
		\end{enumerate}
	\end{frame}
	
	\begin{frame}{Questão 046}	
		\begin{block}{[Q046]}
			É verdade que todo AFN tem um AFD equivalente. Na prova apresentada pelo Sipser, ele constroi um AFD $M = (Q', \Sigma, \delta', q_0', F')$ a partir de um AFN $N = (Q, \Sigma, \delta, q_0, F)$. 
			
			\vspace*{0.3cm}
			
			O estado inicial de M é $E(\{q_0\})$ porque...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item é necessário garantir que os estados que podem ser atingidos a partir de $q_0$ sejam especiais.
			\item é necessário que $q_0$ não seja um conjunto e sim apenas um elemento.
			\item é necessário que apenas $q_0$ seja um estado especial.
			\item é necessário incluir também os estados que podem ser atingidos a partir de $q_0$ ao longo de 0 ou mais setas $\epsilon$.
		\end{enumerate}
	\end{frame}
