		\begin{block}{[Q024]}

		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item 2
			\item 4
			\item não se pode afirmar.
			\item infinitas.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 026}

	\begin{frame}{Questão 029}	
		\begin{block}{[Q029]}
			Na definição formal de computação para um AFD $M$, se $M$ aceita $\omega$, então existe uma sequência de estados $r_0, r_1, \ldots, r_n$ em que
			\begin{itemize}
				\item - $r_0 = q_0$;
				\item - $\delta(r_i, \omega_{i+1}) = r_{i+1}$, para $i=0, \ldots, n-1$, e
				\item - $r_n \in F$.
			\end{itemize}
		O que o valor de $n$ representa
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item a quantidade de estados da sequência
			\item o tamanho da cadeia $\omega$
			\item a quantidade de entradas distintas de $\delta$
			\item a quantidade de estados de $M$
		\end{enumerate}
	\end{frame}
