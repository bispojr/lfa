		\begin{block}{[Q003]}
			Sejam $A$ e $B$ dois conjuntos. Se $|A|=12$ e $|B|= 8$. Se $n = |A \cup B|$, é correto afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $n= 12$
			\item $8 \leq n \leq 12$
			\item $n \leq 8$
			\item $n = 0$	
		\end{enumerate}
