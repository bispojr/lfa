	\begin{frame}{Questão 014}	
		\begin{block}{[Q014]}
			Em relação ao diagrama do autômato finito a seguir, assinale a alternativa que está \underline{incorreta}.
			\begin{center}
				\includegraphics[width=.7\textwidth]{images/afd-m1}
			\end{center}
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $q_1$ é o estado inicial.
			\item $q_2$ é o estado final.
			\item O alfabeto é o conjunto $\{0, 1, 01\}$.
			\item Este diagrama é chamado de diagrama de estados.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 015}

	\begin{frame}{Questão 017}	
		\begin{block}{[Q017]}
			Um autômato finito é definido por uma 5-upla $(Q, \Sigma, \delta, q_0, F)$. Em relação à definição, assinale a alternativa que está \underline{incorreta}.
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $Q$ é um conjunto infinito de estados.
			\item $\Sigma$ é um conjunto finito chamado alfabeto.
			\item $\delta$ é a função de transição.
			\item $F$ é o conjunto de estados finais.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 018}	
		\begin{block}{[Q018]}
			Um autômato finito é definido por uma 5-upla $(Q, \Sigma, \delta, q_0, F)$. A função $\delta$ é definida como se segue
			\begin{center}
				$\delta:Q \times \Sigma \rightarrow Q$
			\end{center}
		Em relação à $\delta$, é \underline{correto} afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item os estados do autômato são necessários apenas no domínio da função.
			\item o contradomínio da função é o alfabeto.
			\item as possibilidades de valores de entradas são infinitas.
			\item é uma função que recebe duas entradas, sendo um estado e um símbolo do alfabeto.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 019}	
		\begin{block}{[Q019]}
			Em relação à linguagem de uma máquina $M$, é \underline{correto} afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $L(M)$ é a notação utilizada para representar que $M$ está ligada.
			\item $L(M)$ é o conjunto de todas as máquinas que têm\\ a linguagem $L$.
			\item é o conjunto de todas as cadeias que a máquina $M$ aceita.
			\item se $L(M) = A$, então $A$ é uma cadeia aceita por $M$.
		\end{enumerate}
	\end{frame}
