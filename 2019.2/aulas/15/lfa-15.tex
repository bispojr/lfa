	\begin{frame}{Questão 062}	
		\begin{block}{[Q062]}
			Seja a gramática livre-do-contexto conforme a descrição abaixo
			\begin{center}
				\begin{itemize}
					\item[] $A \rightarrow$ 0$A$1
					\item[] $A \rightarrow B$
					\item[] $B \rightarrow \#$
				\end{itemize}
			\end{center}
			Qual das cadeias abaixo \underline{não} é gerada pela gramática?
		\end{block}
		
		\begin{enumerate}[label=(\Alph*)]
			\item \#
			\item 0\#1
			\item 01\#01
			\item 00\#11
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 063}	
		\begin{block}{[Q063]}
			Seja a gramática livre-do-contexto conforme a descrição abaixo
			\begin{center}
				\begin{itemize}
					\item[] $S \rightarrow$ a$S$b | $SS$ | $\epsilon$
				\end{itemize}
			\end{center}
			Qual das cadeias abaixo \underline{não} é gerada pela gramática?
		\end{block}
		
		\begin{enumerate}[label=(\Alph*)]
			\item ab
			\item aabb
			\item abab
			\item abba
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 064}	
		\begin{block}{[Q064]}
			Seja a gramática livre-do-contexto conforme a descrição abaixo
			\begin{center}
				\begin{itemize}
					\item[] $\langle$EXPR$\rangle$ $\rightarrow$ $\langle$EXPR$\rangle$+$\langle$TERM$\rangle$ | $\langle$TERM$\rangle$
					\item[] $\langle$TERM$\rangle$ $\rightarrow$ $\langle$TERM$\rangle$x$\langle$FACTOR$\rangle$ | $\langle$FACTOR$\rangle$
					\item[] $\langle$FACTOR$\rangle$ $\rightarrow$ ($\langle$EXPR$\rangle$) | a
				\end{itemize}
			\end{center}
			Qual das cadeias abaixo \underline{não} é gerada pela gramática?
		\end{block}
		
		\begin{enumerate}[label=(\Alph*)]
			\item a+axa
			\item a+x+a
			\item (a+a)xa
			\item axaxa
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 065}	
	\begin{block}{[Q065]}
		A partir da gramática livre-do-contexto descrita abaixo
		\begin{center}
			\begin{itemize}
				\item[] $\langle$EXPR$\rangle$ $\rightarrow$ $\langle$EXPR$\rangle$+$\langle$TERM$\rangle$ | $\langle$TERM$\rangle$
				\item[] $\langle$TERM$\rangle$ $\rightarrow$ $\langle$TERM$\rangle$x$\langle$FACTOR$\rangle$ | $\langle$FACTOR$\rangle$
				\item[] $\langle$FACTOR$\rangle$ $\rightarrow$ ($\langle$EXPR$\rangle$) | a
			\end{itemize}
		\end{center}
		é \underline{incorreto} afirmar que...
	\end{block}
	
	\begin{enumerate}[label=(\Alph*)]
		\item $\langle$FACTOR$\rangle$ $\xRightarrow{\text{*}}$ axa
		\item $\langle$FACTOR$\rangle$ $\xRightarrow{\text{*}}$ (axa)
		\item $\langle$EXPR$\rangle$ $\xRightarrow{\text{*}}$ a
		\item $\langle$EXPR$\rangle$ $\xRightarrow{\text{*}}$ axa
	\end{enumerate}
\end{frame}
	
	\begin{frame}{Questão 066}	
		\begin{block}{[Q066]}
			Sejam duas gramáticas $G_1$ e $G_2$ que geram as linguagens $L_1$ e $L_2$ respectivamente. $G_1$ tem $m$ regras e $G_2$ tem $n$ regras. Reaproveitando as regras existentes em $G_1$ e $G_2$, quantas regras existiriam, no mínimo, na gramática $G$ de forma que $L(G) = L_1 \cup L_2$?
		\end{block}
		
		\begin{enumerate}[label=(\Alph*)]
			\item $m+n+2$
			\item $2(m+n) + 1$
			\item $m \times n + 1$
			\item $m - n + 1$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 067}	
		\begin{block}{[Q067]}
			É \underline{incorreto} afirmar que...
		\end{block}
		
		\begin{enumerate}[label=(\Alph*)]
			\item a classe de linguagens livre-de-contexto é fechada sob a operação de concatenação.
			\item toda linguagem regular é livre-de-contexto.
			\item se uma GLC gera $\omega$, então $\omega$ tem uma única derivação possível.
			\item a classe de linguagens livre-de-contexto é fechada sob a operação estrela.
		\end{enumerate}
	\end{frame}
