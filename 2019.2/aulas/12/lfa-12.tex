	\begin{frame}{Questão 044}	
		\begin{block}{[Q044]}
			É verdade que a classe de linguagens regulares é fechada sob a operação de união. Na prova apresentada pelo Sipser, ele constroi um AFN $N$ a partir de dois AFNs: $N_1$ e $N_2$. A prova mostra que $L(N) = L(N_1) \cup L(N_2)$. 
			
			\vspace{0.3cm}
			Se $N_1$ e $N_2$ têm 20 estados cada um, quantos estados tem $N$?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item 20
			\item 41
			\item $2^{20}$
			\item $40^2$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 045}	
		\begin{block}{[Q045]}
			É verdade que a classe de linguagens regulares é fechada sob a operação de concatenação. Na prova apresentada pelo Sipser, ele constroi um AFN $N$ a partir de dois AFNs: $N_1$ e $N_2$. A prova mostra que $L(N) = L(N_1) \circ L(N_2)$. 
			
			\vspace{0.3cm}
			Se $N_1$ e $N_2$ têm 30 estados cada um, quantos estados tem $N$?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item 15
			\item 30
			\item 60
			\item $2^{30}$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 046}	
		\begin{block}{[Q046]}
			É verdade que a classe de linguagens regulares é fechada sob a operação de estrela. Na prova apresentada pelo Sipser, ele constroi um AFN $N$ a partir do AFN $N_1$. A prova mostra que $L(N) = L(N_1)^*$. 
			
			\vspace{0.3cm}
			Se $N_1$ tem 10 estados, quantos estados tem $N$?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item 5
			\item 9
			\item 10
			\item 11
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 047}	
		\begin{block}{[Q047]}
			É verdade que a classe de linguagens regulares é fechada sob a operação de união. Na prova apresentada pelo Sipser, ele constroi um AFN $N  = (Q, \Sigma, \delta, q_0, F)$ a partir de dois AFNs: $N_1 = (Q_1, \Sigma, \delta_1, q_1, F_1)$ e $N_2 = (Q_2, \Sigma, \delta_2, q_2, F_2)$. A prova mostra que $L(N) = L(N_1) \cup L(N_2)$. 
			
			\vspace{0.3cm}
			Podemos dizer que o valor de $\delta(q_0, \epsilon)$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $\emptyset$
			\item $\{q_1, q_2\}$
			\item $\delta_1(q_0, \epsilon)$
			\item $\delta_2(q_0, \epsilon)$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 048}	
		\begin{block}{[Q048]}
			É verdade que a classe de linguagens regulares é fechada sob a operação de concatenação. Na prova apresentada pelo Sipser, ele constroi um AFN $N  = (Q, \Sigma, \delta, q_1, F)$ a partir de dois AFNs: $N_1 = (Q_1, \Sigma, \delta_1, q_1, F_1)$ e $N_2 = (Q_2, \Sigma, \delta_2, q_2, F_2)$. A prova mostra que $L(N) = L(N_1) \circ L(N_2)$. 
			
			\vspace{0.3cm}
			Se $q \in F_1$, então podemos dizer que o valor de $\delta(q, \epsilon)$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $\delta_1(q, \epsilon)$
			\item $\delta_2(q, \epsilon)$
			\item $\delta_1(q, \epsilon) \cup \{q_2\}$
			\item $\delta_2(q, \epsilon) \cup \{q_2\}$
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 049}	
		\begin{block}{[Q049]}
			É verdade que a classe de linguagens regulares é fechada sob a operação de estrela. Na prova apresentada pelo Sipser, ele constroi um AFN $N  = (Q, \Sigma, \delta, q_0, F)$ a partir do AFN $N_1 = (Q_1, \Sigma, \delta_1, q_1, F_1)$. A prova mostra que $L(N) = L(N_1)^*$. 
			
			\vspace{0.3cm}
			Se $q \in F_1$ e $a \not= \epsilon$, então podemos dizer que o valor de $\delta(q, a)$ é...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item $\delta_1(q, a)$
			\item $\delta_2(q, a)$
			\item $\delta_1(q, a) \cup \{q_1\}$
			\item $\delta_2(q, a) \cup \{q_1\}$
		\end{enumerate}
	\end{frame}
